// ============================================================ // js/auth.js // Funciones de autenticación via Supabase Auth // ============================================================ async function signIn(email, password) { const { data, error } = await supabase.auth.signInWithPassword({ email, password }); return { data, error }; } async function signOut() { await supabase.auth.signOut(); } async function checkSession() { const { data } = await supabase.auth.getSession(); return data?.session || null; }
